25  import java.io.Serializable;
26  
27  import javax.validation.constraints.NotNull;
28  import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
29  
30  import org.apiguardian.api.API;
31  import org.apiguardian.api.API.Status;
32  
33  import com.fasterxml.jackson.annotation.JsonValue;
34  import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
35  
36  /**
37   * Interface representing the identity of a bean. Allows hiding the actual
38   * identity implementation from referencing users.
39   * 
40   * The implementations of this interface must be inmutable.
41   * 
42   * @author <a href="mailto:wamphiry@orne.dev">(w) Iker Hernaez</a>
43   * @version 1.0, 2020-05
44   * @since 0.1
45   */
46  @API(status=Status.STABLE, since="0.1")
47  @XmlJavaTypeAdapter(TokenIdentity.IdentityXmlAdapter.class)
48  @JsonDeserialize(as=TokenIdentity.class)
49  public interface Identity
50  extends Serializable {
51  
52      /**
53       * Returns the identity token. This token must be unique among non equal
54       * instances of the same type of tokens.
55       * 
56       * @return The identity token
57       */
58      @NotNull
59      @ValidIdentityToken
60      @JsonValue
61      String getIdentityToken();
62  
63      /**
64       * Resolves this identity to an identity of the specified target type.
65       * 
66       * @param <T> The target identity type
67       * @param type The target identity type
68       * @return An instance of target identity type
69       * @throws UnrecognizedIdentityTokenException If this identity's
70       * token cannot be resolved to target identity type
71       */
72      default <T extends Identity> @NotNull T resolve(
73              final @NotNull Class<T> type) {
74          return IdentityResolver.getInstance().resolve(this, type);
75      }
76  }
